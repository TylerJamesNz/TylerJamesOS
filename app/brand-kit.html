CSS Tokens

All design tokens are defined in :root at the top of this file. Edit the values in one place and they propagate everywhere.

/* Paste into your app stylesheet — matches :root above (brand hue +30°) */

:root {
  --color-bg: #F7F8F9;
  --color-surface: #ffffff;
  --color-surface-2: #EEF0F2;
  --color-border: #DDE1E5;
  --color-border-soft: #EBEEF0;
  --color-primary: #151B36;
  --color-accent: #193C61;
  --color-accent-hover: #0F2C4B;
  --color-text: #151B36;
  --color-text-muted: #4A5370;
  --color-text-subtle: #8F9AAF;
  --color-info: #193C61;

  --color-sidebar-bg: #151B36;
  --color-sidebar-border: #1E274A;
  --color-sidebar-text: #A8B3CB;
  --color-sidebar-text-hi: #E8EBF2;
  --color-sidebar-accent: #7D8EA0;

  /* Semantic */
  --color-success: #16a34a;
  --color-warning: #d97706;
  --color-destructive: #dc2626;

  /* Fonts */
  --font-display: 'Bricolage Grotesque', sans-serif;   --font-sans: 'Inter', -apple-system, sans-serif;
  --font-mono: 'Geist Mono', monospace;
}
